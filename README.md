

# php-docker
php in docker for arm64 &amp; amd64

Following tags exist in both amd64 && arm64/aarch64:

 - 7-cli-bullseye
 - 7-cli-alpine 
 - 7-fpm-bullseye 
 - 7-fpm-alpine
 - 7-apache-bullseye 
 - 8-cli-bullseye 
 - 8-cli-alpine 
 - 8-apache-bullseye
 - 8-fpm-bullseye
 - 8-fpm-alpine
 - 8.1-fpm-bullseye
 - 8.1-fpm-alpine
 - 8.1-apache-bullseye
 - 8.1-cli-bullseye
 - 8.1-cli-alpine

 - bcmath

 - ctype

 - gd  
 - intl 
 - imagick
 - imap
 - json
 - ldap
 - mbstring
 - mcrypt
 - memcached
 - mongodb
 - mysqli
 - opcache
 - pdo_mysql
 - pdo_pgsql
 - pgsql
 - redis
 - soap
 - sockets
 - tidy
 - timezonedb
 - uuid
 - xsl 
 - yaml
 - zip
 - zstd
